def PCA_SVD(data, n_dim):
    # 导入 numpy 库
    import numpy as np
    
    data_ = data

    # 对数据进行中心化，即每一列减去该列的均值
    data_centered = data_ - np.mean(data_, axis=0)

    # 对中心化后的数据进行奇异值分解（SVD）
    U, S, Vt = np.linalg.svd(data_centered, full_matrices=False)

    # 取前 100 个奇异值对应的奇异向量，构成一个 10000 x 100 的矩阵
    V = Vt.T
    V_reduced = V[:, :n_dim]

    # 将数据投影到这 100 个奇异向量构成的子空间，得到一个 m x 100 的矩阵，即降维后的数据
    data_reduced = np.dot(data_centered, V_reduced)

    # 输出降维后的数据

    #print(data_reduced)
    print(data_reduced.shape)
    #print(V.shape)
    #print(V_reduced.shape)
    
    return data_reduced

def ZNCC(a, b):
    
    # 计算 f 和 w
    f = a.reshape(-1, 1)
    w = b.reshape(-1, 1)

    # 计算 h 和 g
    h = f - np.mean(f)
    g = w - np.mean(w)

    # 计算 gamma
    gamma = (np.dot(h.T, g) / (np.linalg.norm(h) * np.linalg.norm(g)))[0, 0]
    
    return gamma
